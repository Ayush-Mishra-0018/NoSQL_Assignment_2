Instructions to Run Stripe 

1. Required Files

Make sure the following files are present:

Code Files

 `StripeMapFunction.java`
 `StripeInMapper.java`

Supporting File

 `stopwords.txt`

Dataset

 Folder: `Wikipedia-EN-20120601_ARTICLES` 

---

Directory Structure

text
project/
│── StripeMapFunction.java
│── StripeInMapper.java
│── stopwords.txt
│── Wikipedia-EN-20120601_ARTICLES/


---

2. Start Hadoop & Upload Data


start-dfs.sh

hdfs dfs -mkdir -p /user/$USER/input

hdfs dfs -put ../Wikipedia-EN-20120601_ARTICLES /user/$USER/input/

hdfs dfs -put stopwords.txt /user/$USER/


---

1. Merge Dataset 

hdfs dfs -getmerge /user/$USER/input/Wikipedia-EN-20120601_ARTICLES /tmp/merged_input.txt

hdfs dfs -rm -r /user/$USER/input_merged 2>/dev/null
hdfs dfs -mkdir -p /user/$USER/input_merged
hdfs dfs -put /tmp/merged_input.txt /user/$USER/input_merged/


---

1. Compile & Create JARs


javac -classpath `hadoop classpath` -d build_map StripeMapFunction.java
jar -cvf stripe_map.jar -C build_map/ .

javac -classpath `hadoop classpath` -d build_inmap StripeInMapper.java
jar -cvf stripe_inmap.jar -C build_inmap/ .


---

1. Run the Script


sleep 5
hdfs dfsadmin -safemode leave

mkdir -p outputs


---

Run MapFunction


for d in 1 2 3 4; do
    hdfs dfs -rm -r /user/$USER/out_map_d${d} 2>/dev/null

    echo "===== MAP d = ${d} =====" >> runtime_map.txt

    { time HADOOP_OPTS="-Xmx4g" hadoop jar stripe_map.jar StripeMapFunction \
        /user/$USER/input_merged \
        /user/$USER/out_map_d${d} \
        /user/$USER/stopwords.txt \
        ${d}; } 2> time_tmp.txt

    cat time_tmp.txt
    grep -E "^real|^user|^sys" time_tmp.txt >> runtime_map.txt
    echo "" >> runtime_map.txt
done


---

Run In-Mapper Combiner


for d in 1 2 3 4; do
    hdfs dfs -rm -r /user/$USER/out_inmap_d${d} 2>/dev/null

    echo "===== INMAP d = ${d} =====" >> runtime_inmap.txt

    { time HADOOP_OPTS="-Xmx4g" hadoop jar stripe_inmap.jar StripeInMapper \
        /user/$USER/input_merged \
        /user/$USER/out_inmap_d${d} \
        /user/$USER/stopwords.txt \
        ${d}; } 2> time_tmp.txt

    cat time_tmp.txt
    grep -E "^real|^user|^sys" time_tmp.txt >> runtime_inmap.txt
    echo "" >> runtime_inmap.txt
done

rm time_tmp.txt


---

6. Download Outputs


for d in 1 2 3 4; do
    hdfs dfs -getmerge /user/$USER/out_map_d${d} outputs/map_d${d}.txt
    hdfs dfs -getmerge /user/$USER/out_inmap_d${d} outputs/inmap_d${d}.txt
done


---

7. Cleanup


rm /tmp/merged_input.txt
stop-dfs.sh

echo "DONE"


---
